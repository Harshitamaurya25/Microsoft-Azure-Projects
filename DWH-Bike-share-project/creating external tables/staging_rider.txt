IF NOT EXISTS (SELECT * FROM sys.external_file_formats WHERE name = 'SynapseDelimitedTextFormat') 
	CREATE EXTERNAL FILE FORMAT [SynapseDelimitedTextFormat] 
	WITH ( FORMAT_TYPE = DELIMITEDTEXT ,
	       FORMAT_OPTIONS (
			 FIELD_TERMINATOR = ',',
			 USE_TYPE_DEFAULT = FALSE
			))
GO

IF NOT EXISTS (SELECT * FROM sys.external_data_sources WHERE name = 'divvyprojectcontainer_divvprojectstorageacct_dfs_core_windows_net') 
	CREATE EXTERNAL DATA SOURCE [divvyprojectcontainer_divvprojectstorageacct_dfs_core_windows_net] 
	WITH (
		LOCATION = 'abfss://divvyprojectcontainer@divvprojectstorageacct.dfs.core.windows.net', 
		TYPE = HADOOP 
	)
GO

CREATE EXTERNAL TABLE dbo.staging_rider (
	rider_id            NVARCHAR(50),
    first               NVARCHAR(100),
    last                NVARCHAR(100),
    address             NVARCHAR(200),
    birthday            NVARCHAR(50),
    account_start_date  NVARCHAR(50),
    account_end_date    NVARCHAR(50),
    is_member           NVARCHAR(50)
	)
	WITH (
	LOCATION = 'bikeshare/public.rider.csv',
	DATA_SOURCE = [divvyprojectcontainer_divvprojectstorageacct_dfs_core_windows_net],
	FILE_FORMAT = [SynapseDelimitedTextFormat]
	)
GO		


SELECT TOP 100 * FROM dbo.staging_rider
GO
